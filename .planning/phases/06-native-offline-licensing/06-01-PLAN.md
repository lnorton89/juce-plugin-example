    ```cpp
    #include <windows.h>
    #include <sodium.h>
    #include <string>
    #include <array>
    #include <cassert>
    
    std::string deriveMachineIdentifier()
    {
        // 1. Get computer name for SID lookup
        wchar_t computerName[MAX_COMPUTERNAME_LENGTH + 1] = {};
        DWORD nameSize = std::size(computerName);
        if (!GetComputerNameW(computerName, &nameSize))
            return {};
    
        // 2. Get machine SID
        char sidBuffer[SECURITY_MAX_SID_SIZE] = {};
        DWORD sidSize = sizeof(sidBuffer);
        wchar_t domainBuffer[256] = {};
        DWORD domainSize = std::size(domainBuffer);
        SID_NAME_USE sidNameUse;
    
        std::string sidString;
        if (LookupAccountNameW(nullptr, computerName, reinterpret_cast<PSID>(sidBuffer),
                                &sidSize, domainBuffer, &domainSize, &sidNameUse))
        {
            wchar_t* sidStr = nullptr;
            if (ConvertSidToStringSidW(reinterpret_cast<PSID>(sidBuffer), &sidStr))
            {
                int len = WideCharToMultiByte(CP_UTF8, 0, sidStr, -1, nullptr, 0, nullptr, nullptr);
                if (len > 0)
                {
                    sidString.resize(len - 1);
                    WideCharToMultiByte(CP_UTF8, 0, sidStr, -1, sidString.data(), len, nullptr, nullptr);
                }
                LocalFree(sidStr);
            }
        }
    
        // 3. Get system drive volume serial
        wchar_t systemDir[MAX_PATH + 1] = {};
        if (!GetSystemDirectoryW(systemDir, MAX_PATH))
            return {};
        // Extract root path (e.g., "C:\")
        wchar_t rootPath[4] = { systemDir[0], L':', L'\\', L'\0' };
    
        DWORD volumeSerial = 0;
        GetVolumeInformationW(rootPath, nullptr, 0, &volumeSerial,
                              nullptr, nullptr, nullptr, 0);
    
        // 4. Composite: version + SID + volume serial
        const std::string composite = "v1:" + sidString + ":" + std::to_string(volumeSerial);
    
        // 5. SHA-256 hash
        std::array<unsigned char, crypto_hash_sha256_BYTES> hash;
        crypto_hash_sha256(hash.data(),
                           reinterpret_cast<const unsigned char*>(composite.data()),
                           composite.size());
    
        // 6. Base64url encode (manual, no padding)
        static const char alphabet[] = "ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789-_";
        std::string result;
        result.reserve(44);
        for (size_t i = 0; i < hash.size(); i += 3)
        {
            uint32_t triple = (static_cast<uint32_t>(hash[i]) << 16)
                            | (i + 1 < hash.size() ? static_cast<uint32_t>(hash[i + 1]) << 8 : 0)
                            | (i + 2 < hash.size() ? static_cast<uint32_t>(hash[i + 2]) : 0);
            result += alphabet[(triple >> 18) & 0x3F];
            result += alphabet[(triple >> 12) & 0x3F];
            if (i + 1 < hash.size()) result += alphabet[(triple >> 6) & 0x3F];
            if (i + 2 < hash.size()) result += alphabet[triple & 0x3F];
        }
        // Unpadded base64url
        return "mid_v1_" + result;
    }
    ```

    Key details:
    - Call `sodium_init()` at first use — safe to call multiple times per libsodium docs
    - Keep a `static std::once_flag` for one-time init so repeated calls are efficient per LIC-01
    - Never transmit or log raw SID or volume serial per D-01
    - Use `crypto_hash_sha256` (sodium-native) not OpenSSL or manual SHA

    ```cpp
    {
        if (jsonContent.empty() || jsonContent.size() > 64 * 1024)
            return false;
    
        const auto file = resolvePath();
        file.getParentDirectory().createDirectory();
    
        DATA_BLOB input;
        input.pbData = reinterpret_cast<BYTE*>(const_cast<char*>(jsonContent.data()));
        input.cbData = static_cast<DWORD>(jsonContent.size());
    
        DATA_BLOB output;
        if (!CryptProtectData(&input, L"LumaScopeEntitlement", nullptr, nullptr,
                              nullptr, CRYPTPROTECT_UI_FORBIDDEN, &output))
            return false;
    
        juce::FileOutputStream stream(file);
        if (!stream.openedOk())
        {
            LocalFree(output.pbData);
            return false;
        }
    
        const bool ok = stream.write(output.pbData, output.cbData) && stream.flush();
        LocalFree(output.pbData);
        return ok;
    }
    ```

    ```cpp
    std::optional<std::string> LocalEntitlementStore::read()
    {
        const auto file = resolvePath();
        if (!file.existsAsFile())
            return std::nullopt;
    
        juce::FileInputStream stream(file);
        if (!stream.openedOk())
            return std::nullopt;
    
        const auto size = stream.getTotalLength();
        if (size <= 0 || size > 64 * 1024)
            return std::nullopt;
    
        std::vector<BYTE> encryptedData(static_cast<size_t>(size));
        if (stream.read(encryptedData.data(), size) != size)
            return std::nullopt;
    
        DATA_BLOB input;
        input.pbData = encryptedData.data();
        input.cbData = static_cast<DWORD>(encryptedData.size());
    
        DATA_BLOB output;
        if (!CryptUnprotectData(&input, nullptr, nullptr, nullptr,
                                nullptr, CRYPTPROTECT_UI_FORBIDDEN, &output))
            return std::nullopt;  // corrupt, wrong user, or tampered
    
        std::string result(reinterpret_cast<char*>(output.pbData), output.cbData);
        LocalFree(output.pbData);
        return result;
    }
    ```

    ```cpp
    #pragma once
    #include <atomic>
    #include <cstdint>
    #include <juce_core/juce_core.h>
    
    namespace lumascope {
    
    // Thread-safe licensing status — readable from any thread via std::atomic
    // Must never block, allocate, or perform I/O on read
    
    enum class LicenseStatus : std::uint8_t {
        uninitialized = 0,
        not_activated,
        activating,
        activated,
        offline_grace,
        revalidation_required,
        revoked,
        corrupt,
        service_unavailable,
        deactivating
    };
    
    // Human-readable string for bridge events
    const char* toString(LicenseStatus status) noexcept;
    bool isTransitionState(LicenseStatus status) noexcept;
    bool isErrorState(LicenseStatus status) noexcept;
    
    class LicensingState {
    public:
        LicensingState() = default;
        LicensingState(const LicensingState&) = delete;
        LicensingState& operator=(const LicensingState&) = delete;
        
        void updateStatus(LicenseStatus newStatus) noexcept {
            status_.store(newStatus, std::memory_order_release);
            sequence_.fetch_add(1, std::memory_order_release);
        }
        
        LicenseStatus readStatus() const noexcept {
            return status_.load(std::memory_order_acquire);
        }
        
        bool hasChanged(std::uint32_t& lastSeen) const noexcept {
            const auto current = sequence_.load(std::memory_order_acquire);
            if (current == lastSeen) return false;
            lastSeen = current;
            return true;
        }
        
        std::uint32_t currentSequence() const noexcept {
            return sequence_.load(std::memory_order_acquire);
        }
        
    private:
        std::atomic<LicenseStatus> status_{LicenseStatus::uninitialized};
        std::atomic<std::uint32_t> sequence_{0};
    };
    
    } // namespace lumascope
    ```

    **Step 4: Create `plugin/source/Licensing/LicensingCore.cpp`**
    
    Key implementation details:
    
    **Constructor:** store the verifier and store. Don't load from disk yet (deferred to loadFromDisk).
    
    **loadFromDisk():**
    1. Call store_->read()
    2. If nullopt → status = not_activated, return
    3. Parse JSON to SignedEntitlementToken using parseSignedEntitlement()
    4. If parse fails → status = corrupt, return
    5. Verify token with TokenVerifier
    6. If signature invalid → status = corrupt, return
    7. If machineId in token doesn't match current machine → status = corrupt, return
    8. Parse lastVerifiedTime and lastKnownSystemTime from stored metadata
    9. Check grace period: GraceModel::checkOfflineGraceWithRollback()
    10. If expired → status = revalidation_required
    11. If active/warning → status = activated (or offline_grace if elapsed > 0)
    12. Store the parsed metadata
    
    **handleActivationResponse():**
    1. Verify the incoming token (signature, machine match, claims)
    2. If invalid → return, no state change
    3. Save token to store (DPAPI encrypt)
    4. Status → activated
    5. Save lastVerifiedTime = serverTime
    
    **handleDeactivationResponse():**
    1. Per D-11: ONLY clear after server confirms
    2. Call store_->clear()
    3. Status → not_activated
    4. Clear all detail fields

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-6-04 | Tampering | LocalEntitlementStore | mitigate | DPAPI CryptProtectData/CryptUnprotectData binds encrypted blob to the Windows user account. Manual file corruption causes CryptUnprotectData to fail, producing nullopt → corrupt state. Decrypted content is verified via Ed25519 signature by LicensingCore.loadFromDisk(). |
| T-6-05 | Tampering | Clock rollback | mitigate | GraceModel.checkOfflineGraceWithRollback() compares stored lastKnownSystemTime against currentTime. Backward jumps >1 hour trigger rollback path that uses lastKnownSystemTime as the reference point, preventing indefinite grace extension. Documented limitation: sophisticated users who modify time before launch may partially bypass. Per D-07. |
| T-6-06 | Tampering | LicensingCore state machine | mitigate | handleDeactivationResponse() only clears store after server confirms per D-11. handleServerError() preserves stored entitlement per D-12. handleAuthoritativeFailure() transitions to correct error states per D-13. Tests verify each transition. |
| T-6-07 | Information Disclosure | DPAPI encrypted file | accept | File encrypted with current Windows user's credentials. Same-user processes could potentially read it via DPAPI. The signed token contains no PII or raw hardware IDs — only hashed machine ID and license key hash. Risk deemed acceptable per D-03 document limitation. |
| T-6-08 | Denial of Service | Corrupt entitlement file | mitigate | DPAPI decrypt failure → nullopt → corrupt state. File is never overwritten with corrupt data in-place (write creates fresh encrypted blob). No data loss from existing valid state during read failures. |
| T-6-09 | Elevation of Privilege | LicensingCore thread safety | mitigate | LicensingState uses std::atomic with release/acquire ordering. Detail struct protected by mutex (never accessed on audio thread). Audio thread reads atomic status via readStatus() — no locks, no I/O, no allocation per LIC-09. |
